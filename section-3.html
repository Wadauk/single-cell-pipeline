pbmc <- NormalizeData(object = pbmc, normalization.method = "LogNormalize",scale.factor = 10000)
### calculate mito genes precent
mito.genes <- grep("^mt-", rownames(pbmc@data), value = T)
Rpl.genes <- grep("^Rpl", rownames(pbmc@data), value = T)
Mrpl.genes <- grep("^Mrpl", rownames(pbmc@data), value = T)
Rps.genes <- grep("^Rps", rownames(pbmc@data), value = T)
Mrps.genes <- grep("^Mrps", rownames(pbmc@data), value = T)
percent.mito <-colSums(expm1(pbmc@data[c(mito.genes),]))/colSums(expm1(pbmc@data))
percent.ribo <- colSums(expm1(pbmc@data[c(Rpl.genes, Mrpl.genes, Rps.genes,Mrps.genes), ]))/colSums(expm1(pbmc@data))
pbmc <- AddMetaData(pbmc, percent.mito, "percent.mito")
pbmc <- AddMetaData(pbmc, percent.mito, "percent.ribo")
### filter out cell with genes < 200, cell with mito gene percent > 50%
pbmc <- FilterCells(object = pbmc, subset.names = c("nGene","percent.mito"), low.thresholds = c(200, -Inf), high.thresholds = c(3000,0.5))
### filter out gene
pbmc <- FindVariableGenes(object = pbmc, mean.function = ExpMean,dispersion.function = LogVMR, x.low.cutoff = 0.0125, x.high.cutoff = 3,y.cutoff = 0.5,do.plot = FALSE)

3.4.2 聚类合并

for(j in 0:15){
n=table(pbmc@ident)
ident=j
n[names(n) %in% c(0:ident)]=0
n=n[-1]
o=n[n>0]
ident2=names(o[1])
bim <- FindMarkers(pbmc, ident.1 = ident, ident.2 =ident2, only.pos = F, test.use = "bimod")
deg=bim[bim$p_val_adj < 0.01,]
deg1=deg[abs(deg$avg_logFC) >= 1,]
degs=dim(deg)
degs1=dim(deg1)

for(i in c(names(o[-1]))){
bim <- FindMarkers(pbmc, ident.1 = ident,ident.2 = i, only.pos = F, test.use = "bimod")
deg=bim[bim$p_val_adj < 0.01,]
deg1=deg[abs(deg$avg_logFC) >= 1,]
degs=cbind(degs,dim(deg))
degs1=cbind(degs1,dim(deg1))
}
print(degs1)
}
